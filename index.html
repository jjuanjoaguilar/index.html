// proxy-hls.js const express = require('express'); const request = require('request'); const app = express(); const PORT = process.env.PORT || 3000; app.get('/stream.m3u8', (req, res) => { const url = 'https://liveingesta318.cdnmedia.tv/trivisionlive/rtmp01-900/chunklist.m3u8?DVR'; req.pipe(request(url)).pipe(res); }); app.listen(PORT, () => { console.log(`Proxy HLS corriendo en http://localhost:${PORT}`); });
